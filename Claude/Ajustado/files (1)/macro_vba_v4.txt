'============================================================
' PORTAFOLIO OTIC — Macro de Exportación a JSON v4
' COLUMNAS EN "Data Original" (después de eliminar cols 12, 15 y 18):
'   A(1)=No.           B(2)=Estado         C(3)=Proyecto
'   D(4)=SGC/PAA       E(5)=Modalidad      F(6)=Objeto
'   G(7)=Plazo         H(8)=Multiproceso   I(9)=Valor
'   J(10)=F.Abogado    K(11)=Días Abogado
'   L(12)=F.Mesa       M(13)=Días Mesa
'   N(14)=F.Radicación O(15)=Días Radicación
'   P(16)=F.Adjudicación
'   Q(17)=Estructurador R(18)=Abogado S(19)=Evaluadores
'   T(20)=Supervisor    U(21)=Apoyo Supervisión
'
' HOJAS DE TAREAS (Pxx):
'   Col A(1)=#  B(2)=ID  C(3)=Fase  D(4)=Nombre  E(5)=Inicio
'   F(6)=Fin  G(7)=Días  H(8)=Avance  I(9)=Estado
'   Filas 4-18 = tareas, Fila 19 col H = Progreso General
'============================================================

Sub ExportarDashboard()
    Dim wsData As Worksheet
    Dim wsTareas As Worksheet
    Dim jsonStr As String
    Dim filePath As String
    Dim i As Integer, j As Integer
    Dim lastRow As Long
    Dim firstProj As Boolean

    ' Guardar en C:\Temp con timestamp para evitar conflictos
    filePath = "C:\Temp\datos_otic_" & Format(Now(), "HHMMSS") & ".json"
    If Not CreateObject("Scripting.FileSystemObject").FolderExists("C:\Temp") Then
        CreateObject("Scripting.FileSystemObject").CreateFolder "C:\Temp"
    End If

    On Error GoTo ErrorHandler

    Set wsData = ThisWorkbook.Sheets("Data Original")
    lastRow = wsData.Cells(wsData.Rows.Count, 1).End(xlUp).Row

    jsonStr = "{" & vbNewLine
    jsonStr = jsonStr & """fecha_exportacion"": """ & Format(Now(), "YYYY-MM-DD HH:MM:SS") & """," & vbNewLine
    jsonStr = jsonStr & """proyectos"": [" & vbNewLine

    firstProj = True

    For i = 2 To lastRow
        If Trim(CStr(wsData.Cells(i, 1).Value)) <> "" Then

            Dim pid As String
            pid = Trim(CStr(wsData.Cells(i, 1).Value))

            If Not firstProj Then jsonStr = jsonStr & "," & vbNewLine
            firstProj = False

            ' --- CAMPOS DE TEXTO ---
            Dim pEstado As String:        pEstado = LimpiarTexto(CStr(wsData.Cells(i, 2).Value))
            Dim pNombre As String:        pNombre = LimpiarTexto(CStr(wsData.Cells(i, 3).Value))
            Dim pSGC As String:           pSGC = LimpiarTexto(CStr(wsData.Cells(i, 4).Value))
            Dim pModalidad As String:     pModalidad = LimpiarTexto(CStr(wsData.Cells(i, 5).Value))
            Dim pObjeto As String:        pObjeto = LimpiarTexto(CStr(wsData.Cells(i, 6).Value))
            Dim pEstructurador As String: pEstructurador = LimpiarTexto(CStr(wsData.Cells(i, 17).Value))
            Dim pAbogado As String:       pAbogado = LimpiarTexto(CStr(wsData.Cells(i, 18).Value))
            Dim pEvaluadores As String:   pEvaluadores = LimpiarTexto(CStr(wsData.Cells(i, 19).Value))
            Dim pSupervisor As String:    pSupervisor = LimpiarTexto(CStr(wsData.Cells(i, 20).Value))
            Dim pApoyo As String:         pApoyo = LimpiarTexto(CStr(wsData.Cells(i, 21).Value))

            ' --- VALOR (Double para soportar valores grandes) ---
            Dim pValor As Double
            pValor = 0
            If IsNumeric(wsData.Cells(i, 9).Value) Then
                pValor = CDbl(wsData.Cells(i, 9).Value)
            End If

            ' --- FECHAS Y DÍAS ---
            Dim fAbogado As String: fAbogado = FormatFecha(wsData.Cells(i, 10).Value)
            Dim dAbogado As Long:   dAbogado = ValorLong(wsData.Cells(i, 11).Value)
            Dim fMesa As String:    fMesa = FormatFecha(wsData.Cells(i, 12).Value)
            Dim dMesa As Long:      dMesa = ValorLong(wsData.Cells(i, 13).Value)
            Dim fRadic As String:   fRadic = FormatFecha(wsData.Cells(i, 14).Value)
            Dim dRadic As Long:     dRadic = ValorLong(wsData.Cells(i, 15).Value)
            Dim fAdj As String:     fAdj = FormatFecha(wsData.Cells(i, 16).Value)

            ' --- JSON DEL PROYECTO ---
            jsonStr = jsonStr & "  {" & vbNewLine
            jsonStr = jsonStr & "    ""id"": """ & pid & """," & vbNewLine
            jsonStr = jsonStr & "    ""estado"": """ & pEstado & """," & vbNewLine
            jsonStr = jsonStr & "    ""nombre"": """ & pNombre & """," & vbNewLine
            jsonStr = jsonStr & "    ""sgc"": """ & pSGC & """," & vbNewLine
            jsonStr = jsonStr & "    ""modalidad"": """ & pModalidad & """," & vbNewLine
            jsonStr = jsonStr & "    ""objeto"": """ & pObjeto & """," & vbNewLine
            jsonStr = jsonStr & "    ""valor"": " & Format(pValor, "0") & "," & vbNewLine
            jsonStr = jsonStr & "    ""supervisor"": """ & pSupervisor & """," & vbNewLine
            jsonStr = jsonStr & "    ""estructurador"": """ & pEstructurador & """," & vbNewLine
            jsonStr = jsonStr & "    ""abogado"": """ & pAbogado & """," & vbNewLine
            jsonStr = jsonStr & "    ""evaluadores"": """ & pEvaluadores & """," & vbNewLine
            jsonStr = jsonStr & "    ""apoyo"": """ & pApoyo & """," & vbNewLine
            jsonStr = jsonStr & "    ""fecha_abogado"": """ & fAbogado & """," & vbNewLine
            jsonStr = jsonStr & "    ""dias_abogado"": " & dAbogado & "," & vbNewLine
            jsonStr = jsonStr & "    ""fecha_mesa"": """ & fMesa & """," & vbNewLine
            jsonStr = jsonStr & "    ""dias_mesa"": " & dMesa & "," & vbNewLine
            jsonStr = jsonStr & "    ""fecha_radicacion"": """ & fRadic & """," & vbNewLine
            jsonStr = jsonStr & "    ""dias_radicacion"": " & dRadic & "," & vbNewLine
            jsonStr = jsonStr & "    ""fecha_adjudicacion"": """ & fAdj & """," & vbNewLine

            ' --- TAREAS DEL PROYECTO (hoja Pxx) ---
            Dim tareasJson As String: tareasJson = "[]"
            Dim tareaArr As String
            Dim firstTarea As Boolean
            Dim progGen As Long: progGen = 0

            Set wsTareas = Nothing
            On Error Resume Next
            Set wsTareas = ThisWorkbook.Sheets(pid)
            On Error GoTo ErrorHandler

            If Not wsTareas Is Nothing Then
                tareaArr = "["
                firstTarea = True

                For j = 4 To 18
                    Dim tNombreVal As String
                    tNombreVal = Trim(CStr(wsTareas.Cells(j, 4).Value))
                    ' Solo agregar si tiene nombre real (no vacío ni guión)
                    If tNombreVal <> "" And tNombreVal <> "-" Then
                        If Not firstTarea Then tareaArr = tareaArr & ","
                        firstTarea = False

                        Dim tNombre As String: tNombre = LimpiarTexto(tNombreVal)
                        Dim tFase As String:   tFase = LimpiarTexto(CStr(wsTareas.Cells(j, 3).Value))
                        Dim tInicio As String: tInicio = FormatFecha(wsTareas.Cells(j, 5).Value)
                        Dim tFin As String:    tFin = FormatFecha(wsTareas.Cells(j, 6).Value)
                        Dim tDias As Long:     tDias = ValorLong(wsTareas.Cells(j, 7).Value)
                        ' Avance: puede ser 0-100 o decimal 0-1 (formato %)
                        Dim tAvanceRaw As Double: tAvanceRaw = 0
                        Dim tAvance As Long: tAvance = 0
                        If IsNumeric(wsTareas.Cells(j, 8).Value) Then
                            tAvanceRaw = CDbl(wsTareas.Cells(j, 8).Value)
                            If tAvanceRaw > 0 And tAvanceRaw <= 1 Then
                                tAvance = CLng(tAvanceRaw * 100)
                            ElseIf tAvanceRaw > 1 Then
                                tAvance = CLng(tAvanceRaw)
                            End If
                        End If
                        ' Estado: leer texto de celda; si está vacío calcular desde días restantes
                        Dim tEstado As String: tEstado = LimpiarTexto(CStr(wsTareas.Cells(j, 9).Value))
                        If tEstado = "" Or tEstado = "-" Then
                            If tAvance >= 100 Then
                                tEstado = "Completada"
                            ElseIf tDias < 0 Then
                                tEstado = "Vencida"
                            ElseIf tDias <= 5 Then
                                tEstado = "Por vencer"
                            Else
                                tEstado = "En tiempo"
                            End If
                        End If

                        tareaArr = tareaArr & vbNewLine & "      {"
                        tareaArr = tareaArr & """num"": " & (j - 3) & ","
                        tareaArr = tareaArr & """fase"": """ & tFase & ""","
                        tareaArr = tareaArr & """nombre"": """ & tNombre & ""","
                        tareaArr = tareaArr & """inicio"": """ & tInicio & ""","
                        tareaArr = tareaArr & """fin"": """ & tFin & ""","
                        tareaArr = tareaArr & """dias"": " & tDias & ","
                        tareaArr = tareaArr & """avance"": " & tAvance & ","
                        tareaArr = tareaArr & """estado"": """ & tEstado & """"
                        tareaArr = tareaArr & "}"
                    End If
                Next j
                tareasJson = tareaArr & "]"

                ' Progreso general desde celda H19
                ' NOTA: Si la celda está formateada como % el valor real es 0-1 (ej: 0.41 = 41%)
                '       Si la celda contiene 0-100 directamente se usa tal cual
                On Error Resume Next
                If IsNumeric(wsTareas.Cells(19, 8).Value) Then
                    Dim rawProg As Double
                    rawProg = CDbl(wsTareas.Cells(19, 8).Value)
                    ' Si el valor es <= 1 asumimos que es fracción (formato %) y multiplicamos x100
                    If rawProg > 0 And rawProg <= 1 Then
                        progGen = CLng(rawProg * 100)
                    ElseIf rawProg > 1 Then
                        progGen = CLng(rawProg)
                    Else
                        progGen = 0
                    End If
                End If
                On Error GoTo ErrorHandler
            End If

            jsonStr = jsonStr & "    ""progreso_general"": " & progGen & "," & vbNewLine
            jsonStr = jsonStr & "    ""tareas"": " & tareasJson & vbNewLine
            jsonStr = jsonStr & "  }"

            Set wsTareas = Nothing
        End If
    Next i

    jsonStr = jsonStr & vbNewLine & "]" & vbNewLine & "}"

    ' --- ESCRIBIR ARCHIVO ---
    Dim oStream As Object
    Set oStream = CreateObject("ADODB.Stream")
    oStream.Type = 2
    oStream.Charset = "UTF-8"
    oStream.Open
    oStream.WriteText jsonStr
    oStream.SaveToFile filePath, 2
    oStream.Close
    Set oStream = Nothing

    MsgBox "✅ Dashboard exportado correctamente!" & vbNewLine & vbNewLine & _
           "Archivo: " & filePath & vbNewLine & vbNewLine & _
           "Arrastre ese archivo sobre el dashboard_otic.html en el navegador.", _
           vbInformation, "Exportación Exitosa — OTIC"
    Exit Sub

ErrorHandler:
    MsgBox "❌ Error al exportar: " & Err.Description & vbNewLine & _
           "En fila: " & i & ", columna: " & j, vbCritical, "Error de Exportación"
End Sub

' ============================================================
' FUNCIONES AUXILIARES
' ============================================================

Function LimpiarTexto(texto As String) As String
    Dim r As String
    r = texto
    r = Replace(r, Chr(10), ", ")
    r = Replace(r, Chr(13), "")
    r = Replace(r, """", "'")
    r = Replace(r, "\", "/")
    LimpiarTexto = Trim(r)
End Function

Function FormatFecha(valor As Variant) As String
    If IsDate(valor) And Not IsEmpty(valor) And Not IsNull(valor) Then
        FormatFecha = Format(CDate(valor), "YYYY-MM-DD")
    Else
        FormatFecha = ""
    End If
End Function

Function ValorLong(valor As Variant) As Long
    If IsNumeric(valor) And Not IsEmpty(valor) Then
        ValorLong = CLng(valor)
    Else
        ValorLong = 0
    End If
End Function
